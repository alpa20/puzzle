seajs.use(['$', 'popup', 'arale/puzzle/1.0.0/puzzle'], function($, Popup, puzzle) {
  puzzle.init($('#container li'));

  $(document).keyup(function(e) {
    puzzle.push(e.which);
  });

  setTimeout(function() {
    $('#hint').show();
  },1000);

  new Popup({
    trigger: '#hint',
    align: {
      baseXY: ['100%', '50%'],
      selfXY: [0, '50%']
    },
    template: '<div class="hint-popup">支付宝前端类库</div>'
  }).render();
});
